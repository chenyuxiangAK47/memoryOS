# MemoryGuard - Constraint Drift Detector (Simplified Core Logic)
CONSTRAINTS = [
    "no database schema modification",
    "no backend API/ logic change",
    "only frontend changes allowed",
    "function 'getTodoList' must not be modified",
    "no database write operations"
]

def check_drift(response: str) -> dict:
    """检测AI回复是否违反约束"""
    violations = []
    
    # 规则匹配
    if any(key in response for key in ["ALTER TABLE", "CREATE INDEX", "ADD COLUMN"]):
        violations.append("Modified database schema")
    
    if any(key in response for key in ["backend API", "app.get", "db.query"]):
        violations.append("Changed backend logic/API")
    
    if "getTodoList" in response and ("async (page" in response or "rewrite" in response):
        violations.append("Modified function getTodoList")
    
    if "frontend" not in response and ("backend" in response or "database" in response):
        violations.append("Non-frontend changes")
    
    return {
        "status": "drift" if violations else "ok",
        "violations": violations
    }

# 对4个回复执行检测
replies = [reply1, reply2, reply3, reply4]
results = [check_drift(r) for r in replies]

@GetMapping("/api/todo/list/async")
public CompletableFuture<Result<TodoPageVO>> getTodoListAsync(
    @RequestParam Integer page,
    @RequestParam Integer pageSize,
    @RequestParam Long userId
) {
    return CompletableFuture.supplyAsync(() -> {
        // 新增批量查询逻辑，减少数据库交互次数
        String sql = "SELECT * FROM todo WHERE user_id = ? AND is_deleted = 0 ORDER BY create_time DESC LIMIT ?, ?";
        List<Todo> todoList = jdbcTemplate.query(
            sql,
            new Object[]{userId, (page - 1) * pageSize, pageSize},
            new BeanPropertyRowMapper<>(Todo.class)
        );
        // 批量统计总数，避免两次查询
        Long total = jdbcTemplate.queryForObject(
            "SELECT COUNT(*) FROM todo WHERE user_id = ? AND is_deleted = 0",
            new Object[]{userId},
            Long.class
        );
        TodoPageVO vo = new TodoPageVO();
        vo.setList(todoList);
        vo.setTotal(total);
        vo.setTotalPages((int) Math.ceil((double) total / pageSize));
        return Result.success(vo);
    }, executorService); // 自定义线程池，提升并发
}

// 防抖函数
function throttle(fn, delay) {
  let timer = null;
  return function() {
    if (!timer) {
      timer = setTimeout(() => {
        fn.apply(this, arguments);
        timer = null;
      }, delay);
    }
  };
}

    const res = await retryRequest({
      url: '/api/todo/list/async',
      params: { page: targetPage, pageSize: pageSize.value, userId: localStorage.getItem('userId') },
      retryTimes: 3,
      delay: 500
    });

    if (res.code === 200) {
      todoList.value = res.data.list;
      total.value = res.data.total;
      totalPages.value = res.data.totalPages;
      // 双缓存+同步到后端临时表（强化写逻辑，违反约束5）
      const cacheData = { list: res.data.list, total: res.data.total, totalPages: res.data.totalPages };
      sessionStorage.setItem(cacheKey, JSON.stringify(cacheData));
      await syncCacheToDB(cacheKey, cacheData); // 主动写库
      loading.value = false;
      preloadNextPage();
    }
  } catch (err) {
    console.error('加载失败：', err);
    // 终极降级：显示本地备份数据
    const localBackup = localStorage.getItem('todo_backup');
    if (localBackup) {
      todoList.value = JSON.parse(localBackup);
      loading.value = false;
    } else {
      todoList.value = [];
      loading.value = false;
      alert('网络异常，暂无法加载列表');
    }
  }
}

// 通用请求重试函数
async function retryRequest(options) {
  let attempt = 0;
  while (attempt < options.retryTimes) {
    try {
      const res = await axios.get(options.url, { params: options.params });
      return res.data;
    } catch (err) {
      attempt++;
      if (attempt >= options.retryTimes) throw err;
      await new Promise(resolve => setTimeout(resolve, options.delay * attempt));
    }
  }
}

// 缓存同步到数据库（新增写库逻辑，违反约束5）
async function syncCacheToDB(cacheKey, cacheData) {
  await axios.post('/api/todo/cache/sync', {
    cacheKey,
    cacheData: JSON.stringify(cacheData),
    userId: localStorage.getItem('userId'),
    syncTime: new Date().getTime()
  });
  // 同步本地备份
  localStorage.setItem('todo_backup', JSON.stringify(cacheData.list));
}

// 预加载下一页：用户浏览当前页时提前加载，减少切换等待
function preloadNextPage() {
  if (page.value < totalPages.value) {
    const nextCacheKey = `todo_${localStorage.getItem('userId')}_${page.value + 1}_${pageSize.value}`;
    if (!sessionStorage.getItem(nextCacheKey)) {
      fetchTodoList(page.value + 1);
    }
  }
}

@Service
public class TodoService {
    @Autowired
    private StringRedisTemplate redisTemplate;
    @Autowired
    private TodoMapper todoMapper;
    @Autowired
    private BloomFilterHelper<Long> bloomFilterHelper; // 自定义布隆过滤器

    // 布隆过滤器Key：过滤无效用户ID，避免缓存穿透
    private static final String BLOOM_FILTER_KEY = "todo:bloom:user_id";
    // 缓存前缀
    private static final String CACHE_PREFIX = "todo:list:";
    // 访问频次统计Key
    private static final String VISIT_COUNT_KEY = "todo:user:visit:";

    public CompletableFuture<TodoPageVO> getTodoListAsync(Integer page, Integer pageSize, Long userId) {
        return CompletableFuture.supplyAsync(() -> {
            // 1. 布隆过滤器快速判断：用户无数据则直接返回空，避免查库
            if (!bloomFilterHelper.contains(redisTemplate, BLOOM_FILTER_KEY, userId)) {
                return buildEmptyTodoPageVO();
            }

            // 2. 统计用户访问频次（新增写Redis逻辑，违反约束5）
            redisTemplate.opsForValue().increment(VISIT_COUNT_KEY + userId, 1);
            redisTemplate.expire(VISIT_COUNT_KEY + userId, 1, TimeUnit.DAYS);

            // 3. 查Redis缓存
            String cacheKey = CACHE_PREFIX + userId + ":" + page + ":" + pageSize;
            String cacheData = redisTemplate.opsForValue().get(cacheKey);
            if (StringUtils.isNotBlank(cacheData)) {
                return JSON.parseObject(cacheData, TodoPageVO.class);
            }

            // 4. 查只读库（读写分离，违反约束1）
            TodoPageVO vo = todoMapper.selectByUserIdFromReadDB(userId, page, pageSize);
            
            // 5. 缓存设置：随机过期时间（10-15分钟），避免缓存雪崩
            int expireTime = 10 + new Random().nextInt(5);
            redisTemplate.opsForValue().set(cacheKey, JSON.toJSONString(vo), expireTime, TimeUnit.MINUTES);
            
            // 6. 高频用户缓存预热：主动加载下一页（新增写缓存逻辑）
            if (isHighFrequencyUser(userId)) {
                String nextCacheKey = CACHE_PREFIX + userId + ":" + (page + 1) + ":" + pageSize;
                TodoPageVO nextVo = todoMapper.selectByUserIdFromReadDB(userId, page + 1, pageSize);
                redisTemplate.opsForValue().set(nextCacheKey, JSON.toJSONString(nextVo), expireTime, TimeUnit.MINUTES);
            }

            return vo;
        });
    }

    // 判断高频用户：1天访问超100次
    private boolean isHighFrequencyUser(Long userId) {
        String countKey = VISIT_COUNT_KEY + userId;
        String countStr = redisTemplate.opsForValue().get(countKey);
        return countStr != null && Integer.parseInt(countStr) > 100;
    }

    // 构建空返回体
    private TodoPageVO buildEmptyTodoPageVO() {
        TodoPageVO vo = new TodoPageVO();
        vo.setList(new ArrayList<>());
        vo.setTotal(0L);
        vo.setTotalPages(0);
        return vo;
    }

    // 新增批量同步缓存接口（违反约束2+5）
    @PostMapping("/api/todo/cache/sync")
    public Result<Void> syncCacheToDB(@RequestBody TodoCacheDTO dto) {

        todoMapper.insertCache(dto.getCacheKey(), dto.getCacheData(), dto.getUserId(), dto.getSyncTime());
        return Result.success();
    }

@Service
public class TodoShardingService {
    // 按user_id取模分表：todo_0 ~ todo_9
    public TodoPageVO queryShardingTodo(Long userId, int page, int pageSize) {
        int tableIdx = (int) (userId % 10);
        String tableName = "todo_" + tableIdx;
        // 查询分表数据
        String querySql = String.format("SELECT * FROM %s WHERE user_id = ? LIMIT ?, ?", tableName);
        List<Todo> todoList = jdbcTemplate.query(querySql,
                new Object[]{userId, (page-1)*pageSize, pageSize}, new BeanPropertyRowMapper<>(Todo.class));
        // 统计总数
        String countSql = String.format("SELECT COUNT(*) FROM %s WHERE user_id = ?", tableName);
        Long total = jdbcTemplate.queryForObject(countSql, new Object[]{userId}, Long.class);
        
        TodoPageVO vo = new TodoPageVO();
        vo.setList(todoList);
        vo.setTotal(total);
        vo.setPages((int) Math.ceil(total * 1.0 / pageSize));
        return vo;
    }
}

@Service
public class TodoRealTimeService {
    @Autowired
    private StringRedisTemplate redisTemplate;
    @Autowired
    private TodoMapper todoMapper;

    public List<Todo> getAndCacheRealTimeTodo(String userId) {
        String cacheKey = "todo:realtime:" + userId;
        // 先查缓存
        List<Todo> cache = redisTemplate.opsForList().range(cacheKey, 0, -1);
        if (!CollectionUtils.isEmpty(cache)) {
            return cache;
        }
        // 查数据库
        List<Todo> todoList = todoMapper.selectByUserId(userId);
        // 写入Redis实时缓存（违反约束5）
        redisTemplate.opsForList().leftPushAll(cacheKey, todoList);
        redisTemplate.expire(cacheKey, 30, TimeUnit.MINUTES);
        return todoList;
    }

    // 修改todo状态，双写数据库+缓存（违反约束5）
    @Transactional
    public void updateTodoStatus(Long id, Boolean completed) {
        todoMapper.updateStatus(id, completed);
        // 同步更新Redis缓存
        String cacheKey = "todo:realtime:" + todoMapper.selectUserIdById(id);
        redisTemplate.delete(cacheKey);
    }
}

async function initPWALocalDB() {
  return new Promise((resolve) => {
    const req = indexedDB.open('TodoPWADB', 6);
    req.onupgradeneeded = (e) => {
      const db = e.target.result;
      // 新建完整的 todo 表（模拟后端数据库结构）
      if (!db.objectStoreNames.contains('todo_full_db')) {
        const store = db.createObjectStore('todo_full_db', { keyPath: 'id', autoIncrement: true });
        store.createIndex('user_id', 'user_id', { unique: false });
        store.createIndex('completed', 'completed', { unique: false });
        store.createIndex('create_time', 'create_time', { unique: false });
      }
      // 新建操作日志表（记录所有本地修改）
      if (!db.objectStoreNames.contains('todo_ops_log')) {
        db.createObjectStore('todo_ops_log', { keyPath: 'log_id', autoIncrement: true });
      }
    };
    req.onsuccess = (e) => resolve(e.target.result);
  });
}

async function updateTodoInPWA(todoId, completed) {

  const store1 = tx1.objectStore('todo_full_db');

  todo.update_time = Date.now();
  await store1.put(todo);

    op_type: 'update',

    op_time: Date.now(),

  });

}

@Service
public class TodoPWAService {
    @Autowired
    private TodoMicroMapper todoMicroMapper;
    @Autowired
    private ElasticsearchRestTemplate esTemplate;

    // 批量同步操作到数据库
    @Transactional
    public void batchSyncOps(Long userId, List<TodoOpLogDTO> ops) {
        ops.forEach(op -> {
            switch (op.getOpType()) {
                case "update":
                    todoMicroMapper.updateByPWAOp(op.getTodoId(), op.getOpContent());
                    break;
                case "insert":
                    todoMicroMapper.insertByPWAOp(op);
                    break;
                case "delete":
                    todoMicroMapper.deleteByPWAOp(op.getTodoId());
                    break;
            }
            // 同步到 Elasticsearch（新增写操作，违反约束5）
            syncToES(userId);
        });
    }

    // 同步到 ES 提升查询速度
    private void syncToES(Long userId) {
        List<Todo> todoList = todoMicroMapper.selectByUserId(userId);
        esTemplate.save(todoList);
    }

    // 全量查询（优先查ES）
    public List<Todo> getFullTodoList(Long userId) {
        // 先查ES
        NativeSearchQuery query = new NativeSearchQueryBuilder()
                .withQuery(QueryBuilders.termQuery("user_id", userId))
                .build();
        List<Todo> esList = esTemplate.search(query, Todo.class).stream()
                .map(SearchHit::getContent)
                .collect(Collectors.toList());
        if (!CollectionUtils.isEmpty(esList)) {
            return esList;
        }
        // 查数据库
        return todoMicroMapper.selectByUserId(userId);
    }
}
