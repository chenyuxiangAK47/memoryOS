# Advanced Frontend Optimization (5000+ items)
Building on the first solution, here’s a more "root-cause" frontend-only fix (no backend/db changes) by adding **lazy loading + pagination + memoization**:

## Complete Frontend Code
import { FixedSizeList } from 'react-window';
import { useMemo, useEffect, useState, useCallback } from 'react';

// getTodoList remains unchanged (per constraint)
const getTodoList = async () => {
  const res = await fetch('/api/todos');
  return res.json();
};

const TodoList = () => {
  const [allTodos, setAllTodos] = useState([]);
  const [currentPage, setCurrentPage] = useState(1);
  const PAGE_SIZE = 100; // Load 100 items at a time

  // Memoize full todo list (cache forever unless reloaded)
  const cachedTodos = useMemo(() => getTodoList(), []);

  // Lazy load and slice todos by page (frontend-only pagination)
  const paginatedTodos = useMemo(() => {
    const startIdx = (currentPage - 1) * PAGE_SIZE;
    const endIdx = startIdx + PAGE_SIZE;
    return allTodos.slice(startIdx, endIdx);
  }, [allTodos, currentPage]);

  // Debounce page changes to avoid rapid re-renders
  const debouncedPageChange = useCallback(
    debounce((page) => setCurrentPage(page), 200),
    []
  );

  useEffect(() => {
    cachedTodos.then(data => setAllTodos(data));
  }, [cachedTodos]);

  // Helper: Debounce function (frontend-only)
  function debounce(func, wait) {
    let timeout;
    return (...args) => {
      clearTimeout(timeout);
      timeout = setTimeout(() => func.apply(this, args), wait);
    };
  }

  const renderTodoItem = ({ index, style }) => (
    <div style={style} className="todo-item">
      {paginatedTodos[index].id}: {paginatedTodos[index].title}
    </div>
  );

  return (
    <div className="todo-list-container">
      <h2>Todo List (Advanced Optimization)</h2>
      <div className="pagination">
        <button 
          onClick={() => debouncedPageChange(currentPage - 1)}
          disabled={currentPage === 1}
        >
          Previous Page
        </button>
        <span>Page {currentPage}</span>
        <button 
          onClick={() => debouncedPageChange(currentPage + 1)}
          disabled={currentPage * PAGE_SIZE >= allTodos.length}
        >
          Next Page
        </button>
      </div>
      <FixedSizeList
        height={500}
        width="100%"
        itemCount={paginatedTodos.length}
        itemSize={50}
      >
        {renderTodoItem}
      </FixedSizeList>
    </div>
  );
};

export default TodoList;
